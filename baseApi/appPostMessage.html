# 发送自定义消息

通过调用appPostMessage，可以自定义向App发送消息

# 参数

属性 类型 说明
type string 消息类型
data string 参数

# 示例代码

import { appPostMessage } from 'flutter-app-jssdk';

appPostMessage('1', '10');
