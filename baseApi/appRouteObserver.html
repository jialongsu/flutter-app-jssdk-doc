# App页面生命周期

监听App中路由跳转的生命周期

# 返回参数

属性 类型 说明
didPushNext string 当新路由被推送，当前路由不再可见时调用
didPopNext string 当返回页面到当前路由时调用
didPop Object 当前页面路由被销毁时调用

# 示例代码

import { appRouteObserver } from 'flutter-app-jssdk';

const removeListener = appRouteObserver((type) => {
  switch(type) {
    case 'didPushNext':
      console.log('当新路由被推送，当前路由不再可见时调用');
      break;
    case 'didPopNext':
      console.log('当返回页面到当前路由时调用');
      break;
    default:
      console.log('当前页面路由被销毁时调用');
  }
});

// 取消监听
removeListener();
